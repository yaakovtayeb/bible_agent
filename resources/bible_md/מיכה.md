# מיכה

דְּבַראדני אֲשֶׁר הָיָה, אֶלמִיכָה הַמֹּרַשְׁתִּי, בִּימֵי יוֹתָם אָחָז יְחִזְקִיָּה, מַלְכֵי יְהוּדָהאֲשֶׁרחָזָה עַלשֹׁמְרוֹן, וִירוּשָׁלִָם.
שִׁמְעוּ עַמִּים כֻּלָּם, הַקְשִׁיבִי אֶרֶץ וּמְלֹאָהּ וִיהִי אֲדֹנָי אדני בָּכֶם לְעֵד, אֲדֹנָי מֵהֵיכַל קָדְשׁוֹ.
כִּיהִנֵּה אדני, יֹצֵא מִמְּקוֹמוֹ וְיָרַד וְדָרַךְ, עַלבמותי (בָּמֳתֵי) אָרֶץ.
וְנָמַסּוּ הֶהָרִים תַּחְתָּיו, וְהָעֲמָקִים יִתְבַּקָּעוּ, כַּדּוֹנַג מִפְּנֵי הָאֵשׁ, כְּמַיִם מֻגָּרִים בְּמוֹרָד.
בְּפֶשַׁע יַעֲקֹב כָּלזֹאת, וּבְחַטֹּאות בֵּית יִשְׂרָאֵל מִיפֶשַׁע יַעֲקֹב, הֲלוֹא שֹׁמְרוֹן, וּמִי בָּמוֹת יְהוּדָה, הֲלוֹא יְרוּשָׁלִָם.
וְשַׂמְתִּי שֹׁמְרוֹן לְעִי הַשָּׂדֶה, לְמַטָּעֵי כָרֶם וְהִגַּרְתִּי לַגַּי אֲבָנֶיהָ, וִיסֹדֶיהָ אֲגַלֶּה.
וְכָלפְּסִילֶיהָ יֻכַּתּוּ, וְכָלאֶתְנַנֶּיהָ יִשָּׂרְפוּ בָאֵשׁ, וְכָלעֲצַבֶּיהָ, אָשִׂים שְׁמָמָהכִּי מֵאֶתְנַן זוֹנָה, קִבָּצָה, וְעַדאֶתְנַן זוֹנָה, יָשׁוּבוּ.
עַלזֹאת אֶסְפְּדָה וְאֵילִילָה, אֵילְכָה שילל (שׁוֹלָל) וְעָרוֹם אֶעֱשֶׂה מִסְפֵּד כַּתַּנִּים, וְאֵבֶל כִּבְנוֹת יַעֲנָה.
כִּי אֲנוּשָׁה, מַכּוֹתֶיהָכִּיבָאָה, עַדיְהוּדָהנָגַע עַדשַׁעַר עַמִּי, עַדיְרוּשָׁלִָם.
בְּגַת, אַלתַּגִּידוּבָּכוֹ, אַלתִּבְכּוּ בְּבֵית לְעַפְרָה, עָפָר התפלשתי (הִתְפַּלָּשִׁי).
עִבְרִי לָכֶם יוֹשֶׁבֶת שָׁפִיר, עֶרְיָהבֹשֶׁת לֹא יָצְאָה, יוֹשֶׁבֶת צַאֲנָןמִסְפַּד בֵּית הָאֵצֶל, יִקַּח מִכֶּם עֶמְדָּתוֹ.
כִּיחָלָה לְטוֹב, יוֹשֶׁבֶת מָרוֹתכִּייָרַד רָע מֵאֵת אדני, לְשַׁעַר יְרוּשָׁלִָם.
רְתֹם הַמֶּרְכָּבָה לָרֶכֶשׁ, יוֹשֶׁבֶת לָכִישׁ רֵאשִׁית חַטָּאת הִיא לְבַתצִיּוֹן, כִּיבָךְ נִמְצְאוּ פִּשְׁעֵי יִשְׂרָאֵל.
לָכֵן תִּתְּנִי שִׁלּוּחִים, עַל מוֹרֶשֶׁת גַּת בָּתֵּי אַכְזִיב לְאַכְזָב, לְמַלְכֵי יִשְׂרָאֵל.
עֹד, הַיֹּרֵשׁ אָבִי לָךְ, יוֹשֶׁבֶת, מָרֵשָׁהעַדעֲדֻלָּם יָבוֹא, כְּבוֹד יִשְׂרָאֵל.
קָרְחִי וָגֹזִּי, עַלבְּנֵי תַּעֲנוּגָיִךְ הַרְחִבִי קָרְחָתֵךְ כַּנֶּשֶׁר, כִּי גָלוּ מִמֵּךְ

הוֹי חֹשְׁבֵיאָוֶן וּפֹעֲלֵי רָע, עַלמִשְׁכְּבוֹתָם בְּאוֹר הַבֹּקֶר יַעֲשׂוּהָ, כִּי יֶשׁלְאֵל יָדָם.
וְחָמְדוּ שָׂדוֹת וְגָזָלוּ, וּבָתִּים וְנָשָׂאוּ וְעָשְׁקוּ גֶּבֶר וּבֵיתוֹ, וְאִישׁ וְנַחֲלָתוֹ
לָכֵן, כֹּה אָמַר אדני, הִנְנִי חֹשֵׁב עַלהַמִּשְׁפָּחָה הַזֹּאת, רָעָהאֲשֶׁר לֹאתָמִישׁוּ מִשָּׁם צַוְּארֹתֵיכֶם, וְלֹא תֵלְכוּ רוֹמָהכִּי עֵת רָעָה, הִיא.
בַּיּוֹם הַהוּא יִשָּׂא עֲלֵיכֶם מָשָׁל, וְנָהָה נְהִי נִהְיָה אָמַר שָׁדוֹד נְשַׁדֻּנוּחֵלֶק עַמִּי, יָמִיר אֵיךְ יָמִישׁ לִי, לְשׁוֹבֵב שָׂדֵינוּ יְחַלֵּק.
לָכֵן לֹאיִהְיֶה לְךָ, מַשְׁלִיךְ חֶבֶל בְּגוֹרָל, בִּקְהַל, אדני.
אַלתַּטִּפוּ, יַטִּיפוּן לֹאיַטִּפוּ לָאֵלֶּה, לֹא יִסַּג כְּלִמּוֹת.
הֶאָמוּר בֵּיתיַעֲקֹב, הֲקָצַר רוּחַ אדניאִםאֵלֶּה, מַעֲלָלָיו הֲלוֹא דְבָרַי יֵיטִיבוּ, עִם הַיָּשָׁר הֹלֵךְ.
וְאֶתְמוּל, עַמִּי לְאוֹיֵב יְקוֹמֵם, מִמּוּל שַׂלְמָה, אֶדֶר תַּפְשִׁטוּן מֵעֹבְרִים בֶּטַח, שׁוּבֵי מִלְחָמָה.
נְשֵׁי עַמִּי תְּגָרְשׁוּן, מִבֵּית תַּעֲנֻגֶיהָ מֵעַל, עֹלָלֶיהָ, תִּקְחוּ הֲדָרִי, לְעוֹלָם.
קוּמוּ וּלְכוּ, כִּי לֹאזֹאת הַמְּנוּחָה בַּעֲבוּר טָמְאָה תְּחַבֵּל, וְחֶבֶל נִמְרָץ.
לוּאִישׁ הֹלֵךְ רוּחַ, וָשֶׁקֶר כִּזֵּב, אַטִּף לְךָ, לַיַּיִן וְלַשֵּׁכָר וְהָיָה מַטִּיף, הָעָם הַזֶּה.
אָסֹף אֶאֱסֹף יַעֲקֹב כֻּלָּךְ, קַבֵּץ אֲקַבֵּץ שְׁאֵרִית יִשְׂרָאֵליַחַד אֲשִׂימֶנּוּ, כְּצֹאן בָּצְרָהכְּעֵדֶר בְּתוֹךְ הַדָּבְרוֹ, תְּהִימֶנָה מֵאָדָם.
עָלָה הַפֹּרֵץ, לִפְנֵיהֶםפָּרְצוּ וַיַּעֲבֹרוּ, שַׁעַר וַיֵּצְאוּ בוֹ וַיַּעֲבֹר מַלְכָּם לִפְנֵיהֶם, וַאדני בְּרֹאשָׁם

וָאֹמַר, שִׁמְעוּנָא רָאשֵׁי יַעֲקֹב, וּקְצִינֵי, בֵּית יִשְׂרָאֵלהֲלוֹא לָכֶם, לָדַעַת אֶתהַמִּשְׁפָּט.
שֹׂנְאֵי טוֹב, וְאֹהֲבֵי רעה (רָע) גֹּזְלֵי עוֹרָם מֵעֲלֵיהֶם, וּשְׁאֵרָם מֵעַל עַצְמוֹתָם.
וַאֲשֶׁר אָכְלוּ, שְׁאֵר עַמִּי, וְעוֹרָם מֵעֲלֵיהֶם הִפְשִׁיטוּ, וְאֶתעַצְמֹתֵיהֶם פִּצֵּחוּ וּפָרְשׂוּ כַּאֲשֶׁר בַּסִּיר, וּכְבָשָׂר בְּתוֹךְ קַלָּחַת.
אָז יִזְעֲקוּ אֶלאדני, וְלֹא יַעֲנֶה אוֹתָם וְיַסְתֵּר פָּנָיו מֵהֶם בָּעֵת הַהִיא, כַּאֲשֶׁר הֵרֵעוּ מַעַלְלֵיהֶם
כֹּה אָמַר אדני, עַלהַנְּבִיאִים הַמַּתְעִים אֶתעַמִּיהַנֹּשְׁכִים בְּשִׁנֵּיהֶם, וְקָרְאוּ שָׁלוֹם, וַאֲשֶׁר לֹאיִתֵּן עַלפִּיהֶם, וְקִדְּשׁוּ עָלָיו מִלְחָמָה.
לָכֵן לַיְלָה לָכֶם, מֵחָזוֹן, וְחָשְׁכָה לָכֶם, מִקְּסֹם וּבָאָה הַשֶּׁמֶשׁ עַלהַנְּבִיאִים, וְקָדַר עֲלֵיהֶם הַיּוֹם.
וּבֹשׁוּ הַחֹזִים, וְחָפְרוּ הַקֹּסְמִים, וְעָטוּ עַלשָׂפָם, כֻּלָּםכִּי אֵין מַעֲנֵה, אֱלֹהִים.
וְאוּלָם, אָנֹכִי מָלֵאתִי כֹחַ אֶתרוּחַ אדני, וּמִשְׁפָּט, וּגְבוּרָהלְהַגִּיד לְיַעֲקֹב פִּשְׁעוֹ, וּלְיִשְׂרָאֵל חַטָּאתוֹ
שִׁמְעוּנָא זֹאת, רָאשֵׁי בֵּית יַעֲקֹב, וּקְצִינֵי, בֵּית יִשְׂרָאֵלהַמְתַעֲבִים מִשְׁפָּט, וְאֵת כָּלהַיְשָׁרָה יְעַקֵּשׁוּ.
בֹּנֶה צִיּוֹן, בְּדָמִים וִירוּשָׁלִַם, בְּעַוְלָה.
רָאשֶׁיהָ בְּשֹׁחַד יִשְׁפֹּטוּ, וְכֹהֲנֶיהָ בִּמְחִיר יוֹרוּ, וּנְבִיאֶיהָ, בְּכֶסֶף יִקְסֹמוּ וְעַלאדני, יִשָּׁעֵנוּ לֵאמֹר, הֲלוֹא אדני בְּקִרְבֵּנוּ, לֹאתָבוֹא עָלֵינוּ רָעָה.
לָכֵן, בִּגְלַלְכֶם, צִיּוֹן, שָׂדֶה תֵחָרֵשׁ וִירוּשָׁלִַם עִיִּין תִּהְיֶה, וְהַר הַבַּיִת לְבָמוֹת יָעַר

וְהָיָה בְּאַחֲרִית הַיָּמִים, יִהְיֶה הַר בֵּיתאדני נָכוֹן בְּרֹאשׁ הֶהָרִים, וְנִשָּׂא הוּא, מִגְּבָעוֹת וְנָהֲרוּ עָלָיו, עַמִּים.
וְהָלְכוּ גּוֹיִם רַבִּים, וְאָמְרוּ לְכוּ וְנַעֲלֶה אֶלהַראדני וְאֶלבֵּית אֱלֹהֵי יַעֲקֹב, וְיוֹרֵנוּ מִדְּרָכָיו, וְנֵלְכָה בְּאֹרְחֹתָיוכִּי מִצִּיּוֹן תֵּצֵא תוֹרָה, וּדְבַראדני מִירוּשָׁלִָם.
וְשָׁפַט, בֵּין עַמִּים רַבִּים, וְהוֹכִיחַ לְגוֹיִם עֲצֻמִים, עַדרָחוֹק וְכִתְּתוּ חַרְבֹתֵיהֶם לְאִתִּים, וַחֲנִיתֹתֵיהֶם לְמַזְמֵרוֹתלֹאיִשְׂאוּ גּוֹי אֶלגּוֹי חֶרֶב, וְלֹאיִלְמְדוּן עוֹד מִלְחָמָה.
וְיָשְׁבוּ, אִישׁ תַּחַת גַּפְנוֹ וְתַחַת תְּאֵנָתוֹוְאֵין מַחֲרִידכִּיפִי אדני צְבָאוֹת, דִּבֵּר.
כִּי, כָּלהָעַמִּים, יֵלְכוּ, אִישׁ בְּשֵׁם אֱלֹהָיו וַאֲנַחְנוּ, נֵלֵךְ בְּשֵׁםאדני אֱלֹהֵינוּלְעוֹלָם וָעֶד
בַּיּוֹם הַהוּא נְאֻםאדני, אֹסְפָה הַצֹּלֵעָה, וְהַנִּדָּחָה, אֲקַבֵּצָה וַאֲשֶׁר, הֲרֵעֹתִי.
וְשַׂמְתִּי אֶתהַצֹּלֵעָה לִשְׁאֵרִית, וְהַנַּהֲלָאָה לְגוֹי עָצוּם וּמָלַךְ אדני עֲלֵיהֶם בְּהַר צִיּוֹן, מֵעַתָּה וְעַדעוֹלָם
וְאַתָּה מִגְדַּלעֵדֶר, עֹפֶל בַּתצִיּוֹןעָדֶיךָ תֵּאתֶה וּבָאָה, הַמֶּמְשָׁלָה הָרִאשֹׁנָה, מַמְלֶכֶת, לְבַת יְרוּשָׁלִָם.

חוּלִי וָגֹחִי בַּתצִיּוֹן, כַּיּוֹלֵדָהכִּיעַתָּה תֵצְאִי מִקִּרְיָה וְשָׁכַנְתְּ בַּשָּׂדֶה, וּבָאת עַדבָּבֶל שָׁם תִּנָּצֵלִישָׁם יִגְאָלֵךְ אדני, מִכַּף אֹיְבָיִךְ.
