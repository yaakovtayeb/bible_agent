# מלאכי

מַשָּׂא דְבַראדני, אֶליִשְׂרָאֵל, בְּיַד, מַלְאָכִי.
אָהַבְתִּי אֶתְכֶם אָמַר אדני, וַאֲמַרְתֶּם בַּמָּה אֲהַבְתָּנוּ הֲלוֹאאָח עֵשָׂו לְיַעֲקֹב נְאֻםאדני, וָאֹהַב אֶתיַעֲקֹב.
וְאֶתעֵשָׂו, שָׂנֵאתִי וָאָשִׂים אֶתהָרָיו שְׁמָמָה, וְאֶתנַחֲלָתוֹ לְתַנּוֹת מִדְבָּר.
כִּיתֹאמַר אֱדוֹם רֻשַּׁשְׁנוּ, וְנָשׁוּב וְנִבְנֶה חֳרָבוֹתכֹּה אָמַר אדני צְבָאוֹת, הֵמָּה יִבְנוּ וַאֲנִי אֶהֱרוֹס וְקָרְאוּ לָהֶם גְּבוּל רִשְׁעָה, וְהָעָם אֲשֶׁרזָעַם אדני עַדעוֹלָם.
וְעֵינֵיכֶם, תִּרְאֶינָה וְאַתֶּם תֹּאמְרוּ יִגְדַּל אדני, מֵעַל לִגְבוּל יִשְׂרָאֵל.
בֵּן יְכַבֵּד אָב, וְעֶבֶד אֲדֹנָיו וְאִםאָב אָנִי אַיֵּה כְבוֹדִי וְאִםאֲדוֹנִים אָנִי אַיֵּה מוֹרָאִי אָמַר אדני צְבָאוֹת, לָכֶם הַכֹּהֲנִים בּוֹזֵי שְׁמִי, וַאֲמַרְתֶּם, בַּמֶּה בָזִינוּ אֶתשְׁמֶךָ.
מַגִּישִׁים עַלמִזְבְּחִי לֶחֶם מְגֹאָל, וַאֲמַרְתֶּם בַּמֶּה גֵאַלְנוּךָ בֶּאֱמָרְכֶם, שֻׁלְחַן אדני נִבְזֶה הוּא.
וְכִיתַגִּישׁוּן עִוֵּר לִזְבֹּחַ אֵין רָע, וְכִי תַגִּישׁוּ פִּסֵּחַ וְחֹלֶה אֵין רָע הַקְרִיבֵהוּ נָא לְפֶחָתֶךָ, הֲיִרְצְךָ אוֹ הֲיִשָּׂא פָנֶיךָאָמַר, אדני צְבָאוֹת.
וְעַתָּה חַלּוּנָא פְנֵיאֵל, וִיחָנֵּנוּ מִיֶּדְכֶם, הָיְתָה זֹּאתהֲיִשָּׂא מִכֶּם פָּנִים, אָמַר אדני צְבָאוֹת.
מִי גַםבָּכֶם וְיִסְגֹּר דְּלָתַיִם, וְלֹאתָאִירוּ מִזְבְּחִי חִנָּם אֵיןלִי חֵפֶץ בָּכֶם, אָמַר אדני צְבָאוֹת, וּמִנְחָה, לֹאאֶרְצֶה מִיֶּדְכֶם.
כִּי מִמִּזְרַחשֶׁמֶשׁ וְעַדמְבוֹאוֹ, גָּדוֹל שְׁמִי בַּגּוֹיִם, וּבְכָלמָקוֹם מֻקְטָר מֻגָּשׁ לִשְׁמִי, וּמִנְחָה טְהוֹרָהכִּיגָדוֹל שְׁמִי בַּגּוֹיִם, אָמַר אדני צְבָאוֹת.
וְאַתֶּם, מְחַלְּלִים אוֹתוֹבֶּאֱמָרְכֶם, שֻׁלְחַן אֲדֹנָי מְגֹאָל הוּא, וְנִיבוֹ, נִבְזֶה אָכְלוֹ.
וַאֲמַרְתֶּם הִנֵּה מַתְּלָאָה וְהִפַּחְתֶּם אוֹתוֹ, אָמַר אדני צְבָאוֹת, וַהֲבֵאתֶם גָּזוּל וְאֶתהַפִּסֵּחַ וְאֶתהַחוֹלֶה, וַהֲבֵאתֶם אֶתהַמִּנְחָה הַאֶרְצֶה אוֹתָהּ מִיֶּדְכֶם, אָמַר אדני
וְאָרוּר נוֹכֵל, וְיֵשׁ בְּעֶדְרוֹ זָכָר, וְנֹדֵר וְזֹבֵחַ מָשְׁחָת, לַאדֹנָיכִּי מֶלֶךְ גָּדוֹל אָנִי, אָמַר אדני צְבָאוֹת, וּשְׁמִי, נוֹרָא בַגּוֹיִם.

הִנְנִי שֹׁלֵחַ מַלְאָכִי, וּפִנָּהדֶרֶךְ לְפָנָי וּפִתְאֹם יָבוֹא אֶלהֵיכָלוֹ הָאָדוֹן אֲשֶׁראַתֶּם מְבַקְשִׁים, וּמַלְאַךְ הַבְּרִית אֲשֶׁראַתֶּם חֲפֵצִים הִנֵּהבָאאָמַר, אדני צְבָאוֹת.
וּמִי מְכַלְכֵּל אֶתיוֹם בּוֹאוֹ, וּמִי הָעֹמֵד בְּהֵרָאוֹתוֹכִּיהוּא כְּאֵשׁ מְצָרֵף, וּכְבֹרִית מְכַבְּסִים.
וְיָשַׁב מְצָרֵף וּמְטַהֵר, כֶּסֶף, וְטִהַר אֶתבְּנֵילֵוִי וְזִקַּק אֹתָם, כַּזָּהָב וְכַכָּסֶף וְהָיוּ, לַאדני, מַגִּישֵׁי מִנְחָה, בִּצְדָקָה.
וְעָרְבָה, לַאדני, מִנְחַת יְהוּדָה, וִירוּשָׁלִָםכִּימֵי עוֹלָם, וּכְשָׁנִים קַדְמֹנִיֹּת.
וְקָרַבְתִּי אֲלֵיכֶם, לַמִּשְׁפָּט, וְהָיִיתִי עֵד מְמַהֵר בַּמְכַשְּׁפִים וּבַמְנָאֲפִים, וּבַנִּשְׁבָּעִים לַשָּׁקֶר וּבְעֹשְׁקֵי שְׂכַרשָׂכִיר אַלְמָנָה וְיָתוֹם וּמַטֵּיגֵר, וְלֹא יְרֵאוּנִיאָמַר, אדני צְבָאוֹת.
כִּי אֲנִי אדני, לֹא שָׁנִיתִי וְאַתֶּם בְּנֵייַעֲקֹב, לֹא כְלִיתֶם.
לְמִימֵי אֲבֹתֵיכֶם סַרְתֶּם מֵחֻקַּי, וְלֹא שְׁמַרְתֶּםשׁוּבוּ אֵלַי וְאָשׁוּבָה אֲלֵיכֶם, אָמַר אדני צְבָאוֹת וַאֲמַרְתֶּם, בַּמֶּה נָשׁוּב.
הֲיִקְבַּע אָדָם אֱלֹהִים, כִּי אַתֶּם קֹבְעִים אֹתִי, וַאֲמַרְתֶּם, בַּמֶּה קְבַעֲנוּךָהַמַּעֲשֵׂר, וְהַתְּרוּמָה.
בַּמְּאֵרָה אַתֶּם נֵאָרִים, וְאֹתִי אַתֶּם קֹבְעִיםהַגּוֹי, כֻּלּוֹ.
הָבִיאוּ אֶתכָּלהַמַּעֲשֵׂר אֶלבֵּית הָאוֹצָר, וִיהִי טֶרֶף בְּבֵיתִי, וּבְחָנוּנִי נָא בָּזֹאת, אָמַר אדני צְבָאוֹתאִםלֹא אֶפְתַּח לָכֶם, אֵת אֲרֻבּוֹת הַשָּׁמַיִם, וַהֲרִיקֹתִי לָכֶם בְּרָכָה, עַדבְּלִידָי.
וְגָעַרְתִּי לָכֶם בָּאֹכֵל, וְלֹאיַשְׁחִת לָכֶם אֶתפְּרִי הָאֲדָמָה וְלֹאתְשַׁכֵּל לָכֶם הַגֶּפֶן בַּשָּׂדֶה, אָמַר אדני צְבָאוֹת.
וְאִשְּׁרוּ אֶתְכֶם, כָּלהַגּוֹיִםכִּיתִהְיוּ אַתֶּם אֶרֶץ חֵפֶץ, אָמַר אדני צְבָאוֹת
חָזְקוּ עָלַי דִּבְרֵיכֶם, אָמַר אדני וַאֲמַרְתֶּם, מַהנִּדְבַּרְנוּ עָלֶיךָ.
אֲמַרְתֶּם, שָׁוְא עֲבֹד אֱלֹהִים וּמַהבֶּצַע, כִּי שָׁמַרְנוּ מִשְׁמַרְתּוֹ, וְכִי הָלַכְנוּ קְדֹרַנִּית, מִפְּנֵי אדני צְבָאוֹת.
וְעַתָּה, אֲנַחְנוּ מְאַשְּׁרִים זֵדִים גַּםנִבְנוּ עֹשֵׂי רִשְׁעָה, גַּם בָּחֲנוּ אֱלֹהִים וַיִּמָּלֵטוּ.
אָז נִדְבְּרוּ יִרְאֵי אדני, אִישׁ אֶלרֵעֵהוּ וַיַּקְשֵׁב אדני, וַיִּשְׁמָע, וַיִּכָּתֵב סֵפֶר זִכָּרוֹן לְפָנָיו לְיִרְאֵי אדני, וּלְחֹשְׁבֵי שְׁמוֹ.
וְהָיוּ לִי, אָמַר אדני צְבָאוֹת, לַיּוֹם, אֲשֶׁר אֲנִי עֹשֶׂה סְגֻלָּה וְחָמַלְתִּי עֲלֵיהֶםכַּאֲשֶׁר יַחְמֹל אִישׁ, עַלבְּנוֹ הָעֹבֵד אֹתוֹ.
וְשַׁבְתֶּם, וּרְאִיתֶם, בֵּין צַדִּיק, לְרָשָׁעבֵּין עֹבֵד אֱלֹהִים, לַאֲשֶׁר לֹא עֲבָדוֹ
כִּיהִנֵּה הַיּוֹם בָּא, בֹּעֵר כַּתַּנּוּר וְהָיוּ כָלזֵדִים וְכָלעֹשֵׂה רִשְׁעָה, קַשׁ, וְלִהַט אֹתָם הַיּוֹם הַבָּא אָמַר אדני צְבָאוֹת, אֲשֶׁר לֹאיַעֲזֹב לָהֶם שֹׁרֶשׁ וְעָנָף.
וְזָרְחָה לָכֶם יִרְאֵי שְׁמִי, שֶׁמֶשׁ צְדָקָה, וּמַרְפֵּא, בִּכְנָפֶיהָ וִיצָאתֶם וּפִשְׁתֶּם, כְּעֶגְלֵי מַרְבֵּק.
