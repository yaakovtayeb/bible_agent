# צפניה

דְּבַראדני אֲשֶׁר הָיָה, אֶלצְפַנְיָה בֶּןכּוּשִׁי בֶןגְּדַלְיָה, בֶּןאֲמַרְיָה, בֶּןחִזְקִיָּהבִּימֵי יֹאשִׁיָּהוּ בֶןאָמוֹן, מֶלֶךְ יְהוּדָה.
אָסֹף אָסֵף כֹּל, מֵעַל פְּנֵי הָאֲדָמָהנְאֻםאדני.
אָסֵף אָדָם וּבְהֵמָה, אָסֵף עוֹףהַשָּׁמַיִם וּדְגֵי הַיָּם, וְהַמַּכְשֵׁלוֹת, אֶתהָרְשָׁעִים וְהִכְרַתִּי אֶתהָאָדָם, מֵעַל פְּנֵי הָאֲדָמָהנְאֻםאדני.
וְנָטִיתִי יָדִי עַליְהוּדָה, וְעַל כָּליוֹשְׁבֵי יְרוּשָׁלִָם וְהִכְרַתִּי מִןהַמָּקוֹם הַזֶּה, אֶתשְׁאָר הַבַּעַלאֶתשֵׁם הַכְּמָרִים, עִםהַכֹּהֲנִים.
וְאֶתהַמִּשְׁתַּחֲוִים עַלהַגַּגּוֹת, לִצְבָא הַשָּׁמָיִם וְאֶתהַמִּשְׁתַּחֲוִים הַנִּשְׁבָּעִים לַאדני, וְהַנִּשְׁבָּעִים בְּמַלְכָּם.
וְאֶתהַנְּסוֹגִים, מֵאַחֲרֵי אדני וַאֲשֶׁר לֹאבִקְשׁוּ אֶתאדני, וְלֹא דְרָשֻׁהוּ.
הַס, מִפְּנֵי אֲדֹנָי אדניכִּי קָרוֹב יוֹם אדני, כִּיהֵכִין אדני זֶבַח הִקְדִּישׁ קְרֻאָיו.
וְהָיָה, בְּיוֹם זֶבַח אדני, וּפָקַדְתִּי עַלהַשָּׂרִים, וְעַלבְּנֵי הַמֶּלֶךְוְעַל כָּלהַלֹּבְשִׁים, מַלְבּוּשׁ נָכְרִי.
וּפָקַדְתִּי, עַל כָּלהַדּוֹלֵג עַלהַמִּפְתָּןבַּיּוֹם הַהוּאהַמְמַלְאִים בֵּית אֲדֹנֵיהֶם, חָמָס וּמִרְמָה.
וְהָיָה בַיּוֹם הַהוּא נְאֻםאדני, קוֹל צְעָקָה מִשַּׁעַר הַדָּגִים, וִילָלָה, מִןהַמִּשְׁנֶה וְשֶׁבֶר גָּדוֹל, מֵהַגְּבָעוֹת.
הֵילִילוּ, יֹשְׁבֵי הַמַּכְתֵּשׁכִּי נִדְמָה כָּלעַם כְּנַעַן, נִכְרְתוּ כָּלנְטִילֵי כָסֶף
וְהָיָה בָּעֵת הַהִיא, אֲחַפֵּשׂ אֶתיְרוּשָׁלִַם בַּנֵּרוֹת וּפָקַדְתִּי עַלהָאֲנָשִׁים, הַקֹּפְאִים עַלשִׁמְרֵיהֶם, הָאֹמְרִים בִּלְבָבָם, לֹאיֵיטִיב אדני וְלֹא יָרֵעַ.
וְהָיָה חֵילָם לִמְשִׁסָּה, וּבָתֵּיהֶם לִשְׁמָמָה וּבָנוּ בָתִּים, וְלֹא יֵשֵׁבוּ, וְנָטְעוּ כְרָמִים, וְלֹא יִשְׁתּוּ אֶתיֵינָם.
קָרוֹב יוֹםאדני הַגָּדוֹל, קָרוֹב וּמַהֵר מְאֹד קוֹל יוֹם אדני, מַר צֹרֵחַ שָׁם גִּבּוֹר.
יוֹם עֶבְרָה, הַיּוֹם הַהוּאיוֹם צָרָה וּמְצוּקָה, יוֹם שֹׁאָה וּמְשׁוֹאָה, יוֹם חֹשֶׁךְ וַאֲפֵלָה, יוֹם עָנָן וַעֲרָפֶל.
יוֹם שׁוֹפָר, וּתְרוּעָה, עַל הֶעָרִים הַבְּצֻרוֹת, וְעַל הַפִּנּוֹת הַגְּבֹהוֹת.
וַהֲצֵרֹתִי לָאָדָם, וְהָלְכוּ כַּעִוְרִיםכִּי לַאדני, חָטָאוּ וְשֻׁפַּךְ דָּמָם כֶּעָפָר, וּלְחֻמָם כַּגְּלָלִים.
גַּםכַּסְפָּם גַּםזְהָבָם לֹאיוּכַל לְהַצִּילָם, בְּיוֹם עֶבְרַת אדני, וּבְאֵשׁ קִנְאָתוֹ, תֵּאָכֵל כָּלהָאָרֶץכִּיכָלָה אַךְנִבְהָלָה יַעֲשֶׂה, אֵת כָּליֹשְׁבֵי הָאָרֶץ

הוֹי מֹרְאָה, וְנִגְאָלָההָעִיר, הַיּוֹנָה.
לֹא שָׁמְעָה בְּקוֹל, לֹא לָקְחָה מוּסָר בַּאדני לֹא בָטָחָה, אֶלאֱלֹהֶיהָ לֹא קָרֵבָה.
שָׂרֶיהָ בְקִרְבָּהּ, אֲרָיוֹת שֹׁאֲגִים שֹׁפְטֶיהָ זְאֵבֵי עֶרֶב, לֹא גָרְמוּ לַבֹּקֶר.
נְבִיאֶיהָ, פֹּחֲזִים, אַנְשֵׁי, בֹּגְדוֹת כֹּהֲנֶיהָ, חִלְּלוּקֹדֶשׁ, חָמְסוּ, תּוֹרָה.
אדני צַדִּיק בְּקִרְבָּהּ, לֹא יַעֲשֶׂה עַוְלָה בַּבֹּקֶר בַּבֹּקֶר מִשְׁפָּטוֹ יִתֵּן לָאוֹר, לֹא נֶעְדָּר, וְלֹאיוֹדֵעַ עַוָּל, בֹּשֶׁת.
הִכְרַתִּי גוֹיִם, נָשַׁמּוּ פִּנּוֹתָםהֶחֱרַבְתִּי חוּצוֹתָם, מִבְּלִי עוֹבֵר נִצְדּוּ עָרֵיהֶם מִבְּלִיאִישׁ, מֵאֵין יוֹשֵׁב.
אָמַרְתִּי אַךְתִּירְאִי אוֹתִי, תִּקְחִי מוּסָר, וְלֹאיִכָּרֵת מְעוֹנָהּ, כֹּל אֲשֶׁרפָּקַדְתִּי עָלֶיהָ אָכֵן הִשְׁכִּימוּ הִשְׁחִיתוּ, כֹּל עֲלִילוֹתָם.
לָכֵן חַכּוּלִי נְאֻםאדני, לְיוֹם קוּמִי לְעַדכִּי מִשְׁפָּטִי לֶאֱסֹף גּוֹיִם לְקָבְצִי מַמְלָכוֹת, לִשְׁפֹּךְ עֲלֵיהֶם זַעְמִי כֹּל חֲרוֹן אַפִּיכִּי בְּאֵשׁ קִנְאָתִי, תֵּאָכֵל כָּלהָאָרֶץ.
כִּיאָז אֶהְפֹּךְ אֶלעַמִּים, שָׂפָה בְרוּרָה, לִקְרֹא כֻלָּם בְּשֵׁם אדני, לְעָבְדוֹ שְׁכֶם אֶחָד.
מֵעֵבֶר, לְנַהֲרֵיכוּשׁעֲתָרַי, בַּתפּוּצַי, יוֹבִלוּן, מִנְחָתִי.
בַּיּוֹם הַהוּא, לֹא תֵבוֹשִׁי מִכֹּל עֲלִילֹתַיִךְ, אֲשֶׁר פָּשַׁעַתְּ, בִּיכִּיאָז אָסִיר מִקִּרְבֵּךְ, עַלִּיזֵי גַּאֲוָתֵךְ, וְלֹאתוֹסִפִי לְגָבְהָה עוֹד, בְּהַר קָדְשִׁי.
וְהִשְׁאַרְתִּי בְקִרְבֵּךְ, עַם עָנִי וָדָל וְחָסוּ, בְּשֵׁם אדני.
שְׁאֵרִית יִשְׂרָאֵל לֹאיַעֲשׂוּ עַוְלָה, וְלֹאיְדַבְּרוּ כָזָב, וְלֹאיִמָּצֵא בְּפִיהֶם, לְשׁוֹן תַּרְמִיתכִּיהֵמָּה יִרְעוּ וְרָבְצוּ, וְאֵין מַחֲרִיד
רָנִּי, בַּתצִיּוֹןהָרִיעוּ, יִשְׂרָאֵל שִׂמְחִי וְעָלְזִי בְּכָללֵב, בַּת יְרוּשָׁלִָם.
הֵסִיר אדני מִשְׁפָּטַיִךְ, פִּנָּה אֹיְבֵךְ מֶלֶךְ יִשְׂרָאֵל אדני בְּקִרְבֵּךְ, לֹאתִירְאִי רָע עוֹד
בַּיּוֹם הַהוּא, יֵאָמֵר לִירוּשָׁלִַם אַלתִּירָאִיצִיּוֹן, אַליִרְפּוּ יָדָיִךְ.
אדני אֱלֹהַיִךְ בְּקִרְבֵּךְ, גִּבּוֹר יוֹשִׁיעַ יָשִׂישׂ עָלַיִךְ בְּשִׂמְחָה, יַחֲרִישׁ בְּאַהֲבָתוֹיָגִיל עָלַיִךְ, בְּרִנָּה.
נוּגֵי מִמּוֹעֵד אָסַפְתִּי, מִמֵּךְ הָיוּמַשְׂאֵת עָלֶיהָ, חֶרְפָּה.
הִנְנִי עֹשֶׂה אֶתכָּלמְעַנַּיִךְ, בָּעֵת הַהִיא וְהוֹשַׁעְתִּי אֶתהַצֹּלֵעָה, וְהַנִּדָּחָה אֲקַבֵּץ, וְשַׂמְתִּים לִתְהִלָּה וּלְשֵׁם, בְּכָלהָאָרֶץ בָּשְׁתָּם.
בָּעֵת הַהִיא אָבִיא אֶתְכֶם, וּבָעֵת קַבְּצִי אֶתְכֶםכִּיאֶתֵּן אֶתְכֶם לְשֵׁם וְלִתְהִלָּה, בְּכֹל עַמֵּי הָאָרֶץ, בְּשׁוּבִי אֶתשְׁבוּתֵיכֶם לְעֵינֵיכֶם, אָמַר אדני
